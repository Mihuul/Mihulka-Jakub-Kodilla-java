SELECT I.SUMMARY, I.DESCRIPTION, IL.NAME, COUNT(*) AS NUMBER_OF_ISSUES
FROM ISSUES I JOIN ISSUESLISTS IL ON IL.ID = I.ISSUESLIST_ID
GROUP BY IL.NAME;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY, I.DESCRIPTION, COUNT(*) AS JOB_PER_PERSON
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
